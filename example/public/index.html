function concatStr() {
  if (!arguments) throw new Error('No values supplied to concat.');
  var returnString = '';

  for (i = 0; i <= arguments.length; ++i) {
    if (i == arguments.length - 1) {
      return returnString += arguments[i];
    }

    returnString += arguments[i] + ' ';
  }

  return returnString;
}
